---
title: Update Product
api: PUT /api/products/{product}
---

Update an existing product. This endpoint requires admin authentication.

<Warning>
  This endpoint requires admin privileges. Only authenticated users with admin role can update products.
</Warning>

## Authentication

Requires valid Bearer token with admin privileges.

## Path Parameters

<ParamField path="product" type="string" required>
  Product UUID
</ParamField>

## Request Body

All fields are optional. Only include fields you want to update.

<ParamField body="name" type="string">
  Product name (max 255 characters)
</ParamField>

<ParamField body="slug" type="string">
  URL-friendly unique identifier (max 255 characters)
</ParamField>

<ParamField body="category_id" type="string">
  Category UUID (must exist in categories table)
</ParamField>

<ParamField body="team_id" type="string">
  Team UUID (must exist in teams table)
</ParamField>

<ParamField body="price_usd" type="number">
  Price in USD (minimum: 0)
</ParamField>

<ParamField body="price_cop" type="number">
  Price in Colombian Pesos (minimum: 0)
</ParamField>

<ParamField body="description" type="string">
  Product description (nullable)
</ParamField>

<ParamField body="images" type="array">
  Array of image objects or URLs. **Warning:** This replaces all existing images.
  
  <ParamField body="url" type="string" required>
    Image URL
  </ParamField>
  
  <ParamField body="is_primary" type="boolean">
    Whether this is the primary image (default: false)
  </ParamField>
</ParamField>

<ParamField body="variants" type="array">
  Array of size variants. **Warning:** This replaces all existing variants.
  
  <ParamField body="size_id" type="string" required>
    Size UUID (must exist in sizes table)
  </ParamField>
</ParamField>

## Response

<ResponseField name="status" type="string">
  Response status: `success`
</ResponseField>

<ResponseField name="data" type="object">
  Updated product object with relationships loaded (category, team, images, variants, specialCategories)
</ResponseField>

  -H "Content-Type: application/json" \
