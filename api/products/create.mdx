---
title: Create Product
api: POST /api/products
---

Create a new product. This endpoint requires admin authentication.

<Warning>
  This endpoint requires admin privileges. Only authenticated users with admin role can create products.
</Warning>

## Authentication

Requires valid Bearer token with admin privileges.

## Request Body

<ParamField body="name" type="string" required>
  Product name (max 255 characters)
</ParamField>

<ParamField body="slug" type="string" required>
  URL-friendly unique identifier (max 255 characters)
</ParamField>

<ParamField body="category_id" type="string" required>
  Category UUID (must exist in categories table)
</ParamField>

<ParamField body="team_id" type="string" required>
  Team UUID (must exist in teams table)
</ParamField>

<ParamField body="price_usd" type="number" required>
  Price in USD (minimum: 0)
</ParamField>

<ParamField body="price_cop" type="number" required>
  Price in Colombian Pesos (minimum: 0)
</ParamField>

<ParamField body="description" type="string">
  Product description (optional)
</ParamField>

<ParamField body="images" type="array">
  Array of image objects or URLs (optional)
  
  <ParamField body="url" type="string" required>
    Image URL
  </ParamField>
  
  <ParamField body="is_primary" type="boolean">
    Whether this is the primary image (default: false). If no image is marked primary, the first one will be set as primary.
  </ParamField>
</ParamField>

<ParamField body="variants" type="array" required>
  Array of size variants (at least one required). Duplicate sizes are automatically filtered.
  
  <ParamField body="size_id" type="string" required>
    Size UUID (must exist in sizes table)
  </ParamField>
</ParamField>

## Response

<ResponseField name="status" type="string">
  Response status: `success`
</ResponseField>

<ResponseField name="data" type="object">
  Created product object with relationships loaded (category, team, images, variants)
</ResponseField>
