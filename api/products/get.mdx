---
title: Get Product
api: GET /api/products/{product}
---

Retrieve detailed information about a specific product by its ID.

## Path Parameters

<ParamField path="product" type="string" required>
  Product UUID
</ParamField>

## Response

<ResponseField name="status" type="string">
  Response status: `success`
</ResponseField>

<ResponseField name="data" type="object">
  Product object
  
  <ResponseField name="id" type="string">
    Product UUID
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Product name
  </ResponseField>
  
  <ResponseField name="slug" type="string">
    URL-friendly identifier
  </ResponseField>
  
  <ResponseField name="description" type="string">
    Product description
  </ResponseField>
  
  <ResponseField name="price_usd" type="number">
    Price in USD
  </ResponseField>
  
  <ResponseField name="price_cop" type="number">
    Price in Colombian Pesos
  </ResponseField>
  
  <ResponseField name="is_active" type="boolean">
    Whether the product is active
  </ResponseField>
  
  <ResponseField name="category_id" type="string">
    Category UUID
  </ResponseField>
  
  <ResponseField name="team_id" type="string">
    Team UUID
  </ResponseField>
  
  <ResponseField name="category" type="object">
    Category details (id, name, slug)
  </ResponseField>
  
  <ResponseField name="team" type="object">
    Team details including country and league relationships
  </ResponseField>
  
  <ResponseField name="images" type="array">
    Product images
    
    <ResponseField name="id" type="string">
      Image UUID
    </ResponseField>
    
    <ResponseField name="url" type="string">
      Image URL
    </ResponseField>
    
    <ResponseField name="is_primary" type="boolean">
      Whether this is the primary image
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="variants" type="array">
    Product size variants
    
    <ResponseField name="id" type="string">
      Variant UUID
    </ResponseField>
    
    <ResponseField name="size" type="object">
      Size information (id, name, code)
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="special_categories" type="array">
    Special categories (vintage, limited edition, etc.)
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    Creation timestamp
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    Last update timestamp
  </ResponseField>
</ResponseField>

```bash cURL
  -H "Accept: application/json"
```
