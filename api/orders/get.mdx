---
title: "Get Order"
api: "GET /api/orders/{order}"
description: "Retrieve details of a specific order"
authMethod: "bearer"
---

## Authentication

Requires authentication via Bearer token. Users can only view their own orders unless they are admins.

## Path Parameters

<ParamField path="order" type="string" required>
  Order UUID
</ParamField>

## Response

The response includes order details with all relationships loaded:
- Order items with product details, images, category, and team
- Product variant with size information
- Shipping address
- Applied coupon (if any)
- User information (admin only)

<ResponseField name="status" type="string">
  Response status (`success`)
</ResponseField>

<ResponseField name="data" type="object">
  Order object with full details

  <Expandable title="Order Object">
    <ResponseField name="id" type="string">
      Order UUID
    </ResponseField>

    <ResponseField name="user_id" type="string">
      User UUID who placed the order
    </ResponseField>

    <ResponseField name="order_number" type="string">
      Human-readable order number (e.g., `ORD-ABC123`)
    </ResponseField>

    <ResponseField name="subtotal" type="number">
      Subtotal before discounts and fees
    </ResponseField>

    <ResponseField name="tax_amount" type="number">
      Tax amount (currently 0%)
    </ResponseField>

    <ResponseField name="shipping_amount" type="number">
      Shipping cost (currently free)
    </ResponseField>

    <ResponseField name="discount_amount" type="number">
      Discount from coupon
    </ResponseField>

    <ResponseField name="total_amount" type="number">
      Final total amount
    </ResponseField>

    <ResponseField name="status" type="string">
      Order status: `pending`, `paid`, `preparing`, `shipped`, `in_transit`, `delivered`, `cancelled`, `failed`
    </ResponseField>

    <ResponseField name="payment_gateway" type="string">
      Payment gateway used: `wompi` or `paymentnow`
    </ResponseField>

    <ResponseField name="payment_reference" type="string">
      Reference ID from payment gateway
    </ResponseField>

    <ResponseField name="payment_data" type="object">
      Payment gateway-specific data
    </ResponseField>

    <ResponseField name="paid_at" type="string">
      ISO 8601 timestamp when payment was confirmed
    </ResponseField>

    <ResponseField name="shipped_at" type="string">
      ISO 8601 timestamp when order was shipped
    </ResponseField>

    <ResponseField name="delivered_at" type="string">
      ISO 8601 timestamp when order was delivered
    </ResponseField>

    <ResponseField name="notes" type="string">
      Customer notes for the order
    </ResponseField>

    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp when order was created
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp when order was last updated
    </ResponseField>

    <ResponseField name="user" type="object">
      User information (admin only)
    </ResponseField>

    <ResponseField name="items" type="array">
      Array of order items

      <Expandable title="Order Item">
        <ResponseField name="id" type="string">Item UUID</ResponseField>
        <ResponseField name="product_id" type="string">Product UUID</ResponseField>
        <ResponseField name="product_variant_id" type="string">Variant UUID</ResponseField>
        <ResponseField name="product_name" type="string">Product name at time of order</ResponseField>
        <ResponseField name="product_size" type="string">Size name</ResponseField>
        <ResponseField name="unit_price" type="number">Price per unit</ResponseField>
        <ResponseField name="quantity" type="integer">Quantity ordered</ResponseField>
        <ResponseField name="total_price" type="number">Total price for this item</ResponseField>
        <ResponseField name="customization_name" type="string">Player name customization</ResponseField>
        <ResponseField name="customization_number" type="string">Jersey number customization</ResponseField>
        <ResponseField name="product" type="object">Product details with images, category, team</ResponseField>
        <ResponseField name="productVariant" type="object">Variant details with size</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="address" type="object">
      Shipping address

      <Expandable title="Address Fields">
        <ResponseField name="id" type="string">Address UUID</ResponseField>
        <ResponseField name="full_name" type="string">Recipient name</ResponseField>
        <ResponseField name="street_address" type="string">Street address</ResponseField>
        <ResponseField name="city" type="string">City</ResponseField>
        <ResponseField name="state" type="string">State/Province</ResponseField>
        <ResponseField name="postal_code" type="string">Postal code</ResponseField>
        <ResponseField name="country_id" type="string">Country UUID</ResponseField>
        <ResponseField name="phone" type="string">Phone number</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="coupon" type="object">
      Applied coupon details (if any)

      <Expandable title="Coupon Fields">
        <ResponseField name="id" type="string">Coupon UUID</ResponseField>
        <ResponseField name="code" type="string">Coupon code</ResponseField>
        <ResponseField name="type" type="string">Discount type: `percentage` or `fixed`</ResponseField>
        <ResponseField name="value" type="number">Discount value</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET https://api.grancamerino.com/api/orders/9d4e8f2a-1b3c-4d5e-6f7a-8b9c0d1e2f3a \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch(
  'https://api.grancamerino.com/api/orders/9d4e8f2a-1b3c-4d5e-6f7a-8b9c0d1e2f3a',
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN'
    }
  }
);
const data = await response.json();
```
</RequestExample>
